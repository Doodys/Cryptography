public static List<byte> XOR(List<byte> a, List<byte> b)
        {
            List<byte> XOR = new List<byte>();
            for (int i = 0; i < a.Count; i++)
            {
                if ((a[i] == 0 && b[i] == 1) || (a[i] == 1 && b[i] == 0)) { XOR.Add(1); }
                else { XOR.Add(0); }
            }
            return XOR;
        }
